	// Retry with exponential backoff for transient failures
	maxRetries := 0 // 0 retries = 1 attempt total
	var lastErr error

	for attempt := 0; attempt <= maxRetries; attempt++ {
		if attempt > 0 {
			// Exponential backoff: 100ms
			backoffDuration := time.Duration(100*(1<<uint(attempt-1))) * time.Millisecond
			c.log.Warn("retrying GetAccessGraph after failure",
				zap.Int("attempt", attempt),
				zap.Int64("user_id", legacyUserId),
				zap.Duration("backoff", backoffDuration),
				zap.Error(lastErr),
			)
			time.Sleep(backoffDuration)
		}

		graph, err := c.doGetAccessGraph(ctx, legacyUserId)
		if err == nil {
			return graph, nil
		}

		lastErr = err

		// Only retry on timeout/deadline errors and transient server errors
		// Don't retry on 4xx errors (bad user ID, unauthorized, etc.) or 500 (application bug)

		// Check for timeout using type assertion (idiomatic Go approach)
		var netErr net.Error
		isTimeout := errors.As(err, &netErr) && netErr.Timeout()

		// Also check for context deadline exceeded (not always a net.Error)
		if !isTimeout {
			errMsg := err.Error()
			isTimeout = strings.Contains(errMsg, "context deadline exceeded")

		// Check for transient server errors that indicate temporary unavailability:
		// 502 Bad Gateway, 503 Service Unavailable, 504 Gateway Timeout

		if !isTimeout && !isTransientServerError {
